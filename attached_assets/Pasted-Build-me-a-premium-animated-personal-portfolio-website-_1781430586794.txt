STEP 1 — Scene setup:
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(45, 1, 0.1, 100);
  camera.position.z = 4;
  const renderer = new THREE.WebGLRenderer({ 
    canvas: document.getElementById('hero-canvas'),
    alpha: true,        // transparent background
    antialias: true 
  });
  renderer.setPixelRatio(window.devicePixelRatio);
  renderer.setSize(canvasEl.offsetWidth, canvasEl.offsetHeight);

STEP 2 — Create the 3D object (a low-poly icosahedron = "AI Brain"):
  const geometry = new THREE.IcosahedronGeometry(1.3, 1);
  
  // SOLID inner mesh (semi-transparent amber/gold fill):
  const solidMat = new THREE.MeshPhongMaterial({
    color: 0xF5A623,
    transparent: true,
    opacity: 0.08,
    side: THREE.DoubleSide
  });
  const solidMesh = new THREE.Mesh(geometry, solidMat);
  scene.add(solidMesh);
  
  // WIREFRAME outer mesh (dark lines on top):
  const wireMat = new THREE.MeshBasicMaterial({ 
    color: 0x0B0B0B, 
    wireframe: true,
    transparent: true,
    opacity: 0.7
  });
  const wireMesh = new THREE.Mesh(geometry, wireMat);
  scene.add(wireMesh);

STEP 3 — Lighting:
  const ambientLight = new THREE.AmbientLight(0xffffff, 0.8);
  scene.add(ambientLight);
  const pointLight = new THREE.PointLight(0xF5A623, 1.5, 10);
  pointLight.position.set(2, 2, 2);
  scene.add(pointLight);

STEP 4 — Cursor tracking rotation (the signature MetaMask effect):
  let targetRotX = 0, targetRotY = 0;
  let currentRotX = 0, currentRotY = 0;
  
  document.addEventListener('mousemove', (e) => {
    // Map mouse to -1 to +1 range
    targetRotY = ((e.clientX / window.innerWidth) - 0.5) * 2.0;
    targetRotX = ((e.clientY / window.innerHeight) - 0.5) * -1.5;
  });

STEP 5 — Animation loop with lerp (smooth lag = premium feel):
  function animate() {
    requestAnimationFrame(animate);
    
    // Lerp for smooth follow (0.05 = gentle lag, feels alive)
    currentRotX += (targetRotX - currentRotX) * 0.05;
    currentRotY += (targetRotY - currentRotY) * 0.05;
    
    solidMesh.rotation.x = currentRotX;
    solidMesh.rotation.y = currentRotY + Date.now() * 0.0003;
    wireMesh.rotation.x = currentRotX;
    wireMesh.rotation.y = currentRotY + Date.now() * 0.0003;
    
    renderer.render(scene, camera);
  }
  animate();

STEP 6 — Canvas HTML (in hero RIGHT column):

STEP 7 — Orbit particles around the 3D object:
  Add 80 small spheres (radius 0.025) randomly placed in a shell 
  between r=1.8 and r=2.5 around the origin.
  Each particle slowly orbits:
    particle.position.x = Math.cos(time * speed + offset) * radius;
    particle.position.z = Math.sin(time * speed + offset) * radius;
  Use color 0xF5A623 with opacity 0.6.
  This creates the "data cloud" effect around the brain.

STEP 8 — Responsive resize handler:

    renderer.setSize(canvasEl.offsetWidth, canvasEl.offsetHeight);
    camera.aspect = canvasEl.offsetWidth / canvasEl.offsetHeight;
    camera.updateProjectionMatrix();
  });
